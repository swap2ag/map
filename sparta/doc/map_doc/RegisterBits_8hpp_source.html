284  RegisterBits operator>>(uint32_t shift) const
285  {
286  if(num_bytes_ == 8) {
287  return bitShiftRight_<uint64_t>(*this, shift);
288  }
289  else if(num_bytes_ == 4) {
290  return bitShiftRight_<uint32_t>(*this, shift);
291  }
292  else if(num_bytes_ > 8)
293  {
294  RegisterBits final_value(num_bytes_);
295  const uint64_t * src_data = reinterpret_cast<const uint64_t*>(remote_data_);
296  uint64_t * final_data = reinterpret_cast<uint64_t*>(final_value.local_storage_.data());
297  const uint32_t num_dbl_words = num_bytes_ / 8;
298 
299  // Determine the number of double words that will be shifted
300  const uint32_t double_word_shift_count = shift / 64;
301 
302  //
303  // Shift full double words first.
304  //
305  // Example:
306  //
307  // num_dbl_words = 4
308  //
309  // 1111111111111111 2222222222222222 3333333333333333 4444444444444444
310  // 1111111111111111
311  // double_word_shift_count = 3
312  // dest[0] = src[3]
313  //
314  // 1111111111111111 2222222222222222 3333333333333333 4444444444444444
315  // 1111111111111111 2222222222222222
316  // double_word_shift_count = 2
317  // dest[0] = src[2]
318  // dest[1] = src[3]
319  //
320  // 1111111111111111 2222222222222222 3333333333333333 4444444444444444
321  // 1111111111111111 2222222222222222 3333333333333333
322  // double_word_shift_count = 1
323  // dest[0] = src[1]
324  // dest[1] = src[2]
325  // dest[2] = src[3]
326  //
327  uint32_t src_idx = double_word_shift_count;
328  for(uint32_t dest_idx = 0; src_idx < num_dbl_words; ++dest_idx, ++src_idx) {
329  *(final_data + dest_idx) = *(src_data + src_idx);
330  }
331 
332  // Micro-shift:
333  // 1111111111111111 2222222222222222 3333333333333333 4444444444444444
334  // 0011111111111111 1122222222222222 2233333333333333
335  //
336  const auto double_words_to_micro_shift = (num_dbl_words - double_word_shift_count);
337  if(double_words_to_micro_shift > 0)
338  {
339  const uint32_t remaining_bits_to_shift = shift % 64;
340  if(remaining_bits_to_shift)
341  {
342  const uint64_t prev_dbl_word_bits_dropped_mask =
343  (uint64_t)(-(remaining_bits_to_shift != 0) &
344  (-1 >> ((sizeof(uint64_t) * 8) - remaining_bits_to_shift)));
345  const uint32_t prev_dbl_word_bit_pos = 64 - remaining_bits_to_shift;
346  uint32_t idx =0;
347  while(true)
348  {
349  *(final_data + idx) >>= remaining_bits_to_shift;
350  if(++idx == double_words_to_micro_shift) {
351  break;
352  }
353  const uint64_t bits_pulled_in =
354  (*(final_data + idx) & prev_dbl_word_bits_dropped_mask) << prev_dbl_word_bit_pos;
355  *(final_data + (idx - 1)) |= bits_pulled_in;
356  }
357  }
358  }
359  return final_value;
360  }
361  else if(num_bytes_ == 2) {
362  return bitShiftRight_<uint16_t>(*this, shift);
363  }
364  else if(num_bytes_ == 1) {
365  return bitShiftRight_<uint8_t>(*this, shift);
366  }
367 
368  return RegisterBits(nullptr);
369  }
